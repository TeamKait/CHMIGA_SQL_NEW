>ЗАПРОС 1
```sql
SELECT
	concat(l_name, " ", f_name) AS "ФИ",
	count(*) AS "Количество"
FROM adventures
	GROUP BY f_name, l_name
	ORDER BY count(*)
	DESC
	LIMIT 1
```

>ЗАПРОС 2
```sql
SELECT
	town_name AS "Город",
	count(*) AS "Посещения"
FROM(
	SELECT town1 AS town_name FROM adventures
	UNION ALL
	SELECT town2 AS town_name FROM adventures
) AS all_towns
	GROUP BY town_name
	ORDER BY "Посещения" DESC;
```

>ЗАПРОС 3
```sql
SELECT
	adventure_date AS "День",
	count(*) AS "Популярность"
FROM adventures
	GROUP BY adventure_date
	ORDER BY count(*) DESC
	LIMIT 1;
```
